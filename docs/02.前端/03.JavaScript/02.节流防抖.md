/**
 * 节流
 * @param {Function} callback 外部方法
 * @param {Integer} waitTime 等待时间
 * */
export function throttle(callback, waitTime) {
	let flag = false;
	let timer = null;
	return function() {
		// 没有执行前不做
		if(!flag) {
			flag = true;
			// 立即执行
			typeof callback === 'function' && callback();
			timer  = setTimeout(() => {
				// 非立即执行
				// typeof callback === 'function' && callback();
				flag = false;
				timer = null;
			}, waitTime);
		}
	}
}

/**
 * 防抖
 * @param {Function} callback 外部方法
 * @param {Integer} waitTime 等待时间
 * */
export function debounce(callback, waitTime) {
	let timer = null;
	return function() {
		// 点击重置事件，重新计时
		if(timer != null){
			clearTimeout(timer);
		}
		timer  = setTimeout(() => {
			if(callback) callback();
			timer = null;
		}, waitTime);
	}
}
